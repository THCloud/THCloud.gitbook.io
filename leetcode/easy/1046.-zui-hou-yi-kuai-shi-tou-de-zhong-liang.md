# 1046. 最后一块石头的重量

{% embed url="https://leetcode.cn/problems/last-stone-weight/" %}

```
// 模拟
class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        std::priority_queue<int> q(stones.begin(), stones.end());
        while (q.size() > 1) {
            auto s1 = q.top();
            q.pop();
            auto s2 = q.top();
            q.pop();
            if (s1 > s2) q.push(s1 - s2);
        }
        return q.size() == 1 ? q.top() : 0;
    }
};
```

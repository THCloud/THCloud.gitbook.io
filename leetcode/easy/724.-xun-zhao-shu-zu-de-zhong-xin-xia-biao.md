# 724. 寻找数组的中心下标

{% embed url="https://leetcode.cn/problems/find-pivot-index/" %}

```
// 计算数组左右侧累加和
class Solution {
public:
    int pivotIndex(vector<int>& nums) {
        map<int, int> m;
        int res = -1;
        int cur = 0;
        for (int i = nums.size() - 2; i >= 0; --i) {
            cur += nums[i+1];
            m[i] = cur;
        }
        cur = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (i > 0) cur += nums[i - 1];
            if (m[i] == cur) return i;
        }
        return res;
    }
};
```

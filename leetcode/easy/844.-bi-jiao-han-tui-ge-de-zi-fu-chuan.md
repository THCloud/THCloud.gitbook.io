# 844. 比较含退格的字符串

{% embed url="https://leetcode.cn/problems/backspace-string-compare/" %}

```
// 直接模拟过程即可
class Solution {
public:
    bool backspaceCompare(string s, string t) {
        return foo(s) == foo(t);
    }

    string foo(string s) {
        string res;
        int cnt = 0;
        for (int i = s.size() - 1; i >= 0; --i) {
            if (s[i] == '#') cnt++;
            else {
                if (cnt) --cnt;
                else res.push_back(s[i]);
            }
        }
        return res;
    }
};
```
